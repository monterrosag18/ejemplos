# ------------------------------------------------------------
# CRUD de Inventario con Diccionarios
# ------------------------------------------------------------
# Cada producto se guarda con un ID √∫nico como clave.
# El valor es un diccionario con nombre y cantidad.
# ------------------------------------------------------------

inventario = {
    1: {"nombre": "Laptop", "cantidad": 10},
    2: {"nombre": "Mouse", "cantidad": 50},
    3: {"nombre": "Teclado", "cantidad": 30}
}

def crear_producto(dic, id, nombre, cantidad):
    """
    Crea un producto nuevo en el inventario.
    Validaciones:
    - ID √∫nico y entero positivo.
    - Nombre no vac√≠o.
    - Cantidad entero positivo.
    """
    if id in dic:
        print("‚ùå ID duplicado.")
        return
    if not isinstance(nombre, str) or not nombre.strip():
        print("‚ùå Nombre inv√°lido.")
        return
    if not isinstance(cantidad, int) or cantidad < 0:
        print("‚ùå Cantidad inv√°lida.")
        return
    dic[id] = {"nombre": nombre.strip(), "cantidad": cantidad}
    print(f"‚úÖ Producto creado: {dic[id]}")

def leer_inventario(dic):
    """Muestra todos los productos."""
    for id, datos in dic.items():
        print(f"ID={id} | {datos['nombre']} | Cantidad={datos['cantidad']}")

def actualizar_producto(dic, id, nuevo_nombre=None, nueva_cantidad=None):
    """Actualiza nombre y/o cantidad de un producto existente."""
    if id not in dic:
        print("‚ùå Producto no encontrado.")
        return
    if nuevo_nombre:
        dic[id]["nombre"] = nuevo_nombre.strip()
    if nueva_cantidad is not None:
        if nueva_cantidad < 0:
            print("‚ùå Cantidad inv√°lida.")
            return
        dic[id]["cantidad"] = nueva_cantidad
    print(f"‚úèÔ∏è Producto actualizado: {dic[id]}")

def eliminar_producto(dic, id):
    """Elimina un producto por ID."""
    if id not in dic:
        print("‚ùå Producto no encontrado.")
        return
    del dic[id]
    print(f"üóëÔ∏è Producto con ID={id} eliminado.")

# DEMO
leer_inventario(inventario)
crear_producto(inventario, 4, "Monitor", 15)
actualizar_producto(inventario, 2, nueva_cantidad=60)
eliminar_producto(inventario, 3)
leer_inventario(inventario)
